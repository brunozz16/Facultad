import java.util.Calendar;

public abstract class Trabajador {
	private String nombre;
	private String apellido;
	private String direccion;
	private Integer dni;

	public static enum tipo{Jefe,Jornalizado,Mensualizado};

	public Trabajador(String nombre, String apellido, String direccion, Integer dni) {
		super();
		this.nombre = nombre;
		this.apellido = apellido;
		this.direccion = direccion;
		this.dni = dni;
	}

	public abstract float calcularSueldo(Calendar periodo);

	public String toString() {
		if(esDelTipo(tipo.Jefe)) {
			return "Jefe ["+nombre + ", " + apellido + ", direccion=" + direccion + ", dni=" + dni
					+  ", sueldo=$ " ;
		}
		else {
			return "Trabajador ["+nombre + ", " + apellido + ", direccion=" + direccion + ", dni=" + dni
					+  ", sueldo=$ " ;
		}


	}
	public Boolean esDelTipo(Trabajador.tipo tipo) {
		boolean respuesta=false;
		if(tipo.equals(Trabajador.tipo.Jefe)) {
			if(this instanceof Jefe) respuesta=true;
		}
		return respuesta; 
	}

}
