package ejer3;

import java.util.ArrayList;
import java.util.Collection;

public class Carrera {
	private String nombre;
	private Collection<Alumno> alumnos;

	public Carrera(String name) {
		this.nombre=name;
		alumnos=new ArrayList<Alumno>();
	}
	public String toString() {
		return this.nombre;
	}
	public void CargarAlumno(Alumno alum) {
		alumnos.add(alum);
	}
	public void mostrarAlumnos() {
		for(Alumno obj:this.alumnos) {
			System.out.println(obj.toString());
		}
	}
	
}

package ejer3;

import java.util.Calendar;

public class Alumno {
	private Calendar fecha_inscripcion;
	private Persona persona;
	private Carrera carrera;
	
	public Alumno(Integer day,Integer month,Integer anno, Persona persona, Carrera carrera) {
		super();
		fecha_inscripcion=fecha_inscripcion.getInstance();
		fecha_inscripcion.set(Calendar.YEAR, anno);
		fecha_inscripcion.set(Calendar.MONTH,month);
		fecha_inscripcion.set(Calendar.DAY_OF_MONTH,day);
		this.fecha_inscripcion = fecha_inscripcion;
		this.persona = persona;
		this.carrera = carrera;
	}
	public String toString(){ 
		return this.persona.toString()+" - "+
			fecha_inscripcion.get(Calendar.DAY_OF_MONTH)+"/"+
			fecha_inscripcion.get(Calendar.MONTH)+"/"+
			fecha_inscripcion.get(Calendar.YEAR);
	}
}

public class Facultad {

		this.nombre = nombre;
		carreras=new ArrayList<Carrera>();

		return this.nombre;

		carreras.add(car);

	public void mostrarCarrerasyAlumnos() {
	
	System.out.print("Facultad ");

		System.out.print("Carrera: ");
		System.out.println(obj.mostrarNombre());
		System.out.println("Alumnos: ");

	}
	}
}
